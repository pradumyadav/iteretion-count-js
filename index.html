Q1> Find the number of digits?

const Find_Digits = (N) =>

{

let ans=0;

while(N!==0)

{

ans++;

N=Math.floor(N/10);

}

return ans;

};

Q2>Find the Fives.

const Find_Five = (N) =>

{

let res=0;

while(N!==0){

if(N%10===5){

res++;

}

N=Math.floor(N/10);

}

return res;

};

Q7> Check whether a Number is a prime or not.

const Prime_Check = (n) =>

{

let factor =0;

for(i =1;i<=n;i++) {

if(n%i===0){

factor++;

}

}

if(factor===2){

}

else {

}

};
